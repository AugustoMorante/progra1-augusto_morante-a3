Algoritmo Ejercicio1
	Leer C
	m5<-trunc(C/5)
	m2<-trunc((C MOD 5)/2)
	m1<-trunc((C MOD 5) MOD 2)
	Escribir "Monedas de 5"
	Escribir m5
	Escribir "Monedas de 2"
	Escribir m2
	Escribir "Monedas de 1"
	Escribir m1
FinAlgoritmo
